33/// Executes a command and provides real-time visual feedback while processing.
34///
35/// This function displays a spinner animation while a command is executed. If the command
36/// succeeds, the spinner stops with a success message. If the command fails, the spinner
37/// stops with the error output displayed and an error is returned.
38///
39/// # Arguments
40///
41/// * `message` - A message to be displayed alongside the spinner while the command runs.
42/// * `cmd` - A mutable reference to a `Command` object that specifies the process to be executed.
43/// * `success` - A string message displayed when the command executes successfully.
44/// * `failure` - A string error message returned when the command fails.
45/// * `file` - A file name used to store the command's stdout and stderr logs under the `.breathes` directory.
46///
47/// # Returns
48///
49/// Returns `Ok(())` if the command executes successfully, or `Err(std::io::Error)` if the command fails.
50///
51/// # Side Effects
52///
53/// - The function creates two files in the `.breathes` directory:
54///   - `stdout/{file}`: Stores the standard output of the command.
55///   - `stderr/{file}`: Stores the standard error of the command.
56/// - If these files cannot be created or the command cannot be executed, the function panics with an appropriate error message.
57///
58/// # Panics
59/// if command is not founded and if the process cannot be spawned.
60/// # Errors
61///
62/// - Returns `Err(std::io::Error)` with the failure message if the command exits unsuccessfully.
63/// - Returns `Err(std::io::Error)` if there is an issue reading the command's stderr output.
64///
65/// # Examples
66///
67/// ```no_run
68/// use std::process::Command;
69/// use std::io;
70///
71/// fn main() -> io::Result<()> {
72///     let mut cmd = Command::new("ls");
73///     ok(
74///         "Executing command...",
75///         &mut cmd,
76///         "Command executed successfully!",
77///         "Command execution failed.",
78///         "output.log",
79///     )
80/// }
81/// ```
82pub fn ok(message: &str, cmd: &mut Command, success: &str, failure: &str) -> Result<(), Error> {
83    let mut output = Spinner::new(Spinners::Line, message.white().to_string());
84    let status = cmd
85        .current_dir(".")
86        .stdout(Stdio::null())
87        .stderr(Stdio::null())
88        .spawn()?
89        .wait()?
90        .code();
91    if let Some(response) = status
92        && response.eq(&0)
93    {
94        output.stop_and_persist(
95            "âœ“".green().to_string().as_str(),
96            success.dark_cyan().to_string(),
97        );
98        Ok(())
99    } else {
100        output.stop_and_persist("!".red().to_string().as_str(), failure.yellow().to_string());
101        Err(Error::other(failure))
102    }
103}
104///
105/// Call git or mercurial with arg
106///
107/// # Errors
108///
109/// On failure to execute the command
110///
111pub fn call(program: &str, arg: &str) -> Result<i32, Error> {
112    let to = if arg.is_empty() {
113        program.to_string()
114    } else {
115        format!("{program} {arg}")
116    };
117    if !Command::new("sh")
118        .arg("-c")
119        .arg(to.as_str())
120        .current_dir(".")
121        .spawn()?
122        .wait()?
123        .success()
124    {
125        return Err(Error::other(format!("{program} not founded")));
126    }
127    Ok(OK)
128}

206///
207/// Interact with the developer.
208///
209/// # Panics
210/// On bad input
211/// # Errors
212/// - If there is an issue reading the command's stderr output.
213/// - If there is an issue executing a command.
214/// - If there is an issue creating a file.
215/// - If there is an issue clearing the terminal screen.
216/// - If there is an issue writing to a file.
217/// # Dependencies
218/// - `crossterm` for terminal manipulation.
219/// - `git` for version control.
220/// - `git-commit-template` for commit templates.
221///
222pub fn zen() -> Result<i32, InquireError> {
223    loop {
224        execute!(stdout(), Clear(ClearType::All), MoveTo(0, 0))?;
225        let option = Select::new(
226            "@wishes:".green().bold().to_string().as_str(),
227            ZenOption::all(),
228        )
229        .prompt()?;
230        let response: Result<i32, Error> = match option {
231            ZenOption::Quit => {
232                execute!(stdout(), Clear(ClearType::All), MoveTo(0, 0))?;
233                exit(QUIT);
234            }
235            ZenOption::Add => call(vcs().as_str(), "add ."),
236            ZenOption::Health => run_hooks(),
237            ZenOption::Log => call(vcs().as_str(), "log"),
238            ZenOption::Status => call(vcs().as_str(), "status"),
239            ZenOption::Diff => call(vcs().as_str(), "diff"),
240            ZenOption::Email => call("aerc", ""),
241            ZenOption::ListTags => call(vcs().as_str(), "tag"),
242            ZenOption::Edit => call("broot", "."),
243            ZenOption::Commit => {
244                if run_hooks().is_ok()
245                    && let Ok(c) = Commit::default().commit()
246                    && run_commit(c).is_ok()
247                {
248                    call(vcs().as_str(), "push")
249                } else {
250                    continue;
251                }
252            }
253        };
254        if response.is_err() && Confirm::new("Exit").with_default(false).prompt()?
255            || response.is_ok() && Confirm::new("Exit").with_default(false).prompt()?
256        {
257            exit(QUIT);
258        }
259    }
