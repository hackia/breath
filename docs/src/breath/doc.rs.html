7/// check if a command it's safe to execute
8/// # Panics
9/// if the regex is not valid
10#[must_use]
11pub fn is_safe_command(cmd: &str) -> bool {
12    let re = Regex::new(r"([;&|`$]|\.\.)").expect("invalid regex for is_safe_command");
13    let starts_with_slash = cmd.trim().starts_with('/');
14    let allowed_bins = ["cargo", "doxygen", "rm", "cp", "mkdir", "touch"];
15    let binary_name = cmd.split_whitespace().next().unwrap_or("");
16    let is_allowed_bin = allowed_bins.contains(&binary_name);
17    !re.is_match(cmd) && !starts_with_slash && is_allowed_bin
18}

20/// run a command and describe it with success and failure messages
21///
22/// # Errors
23/// on no founded command or bad user input
24pub fn describe(it: &str, command: &str, success: &str, failure: &str) -> Result<(), Error> {
25    if !is_safe_command(command) {
26        return Err(Error::other("Security check failed"));
27    }
28    if cfg!(windows) {
29        ok(
30            it,
31            Command::new("cmd").arg("/C").arg(command),
32            success,
33            failure,
34        )
35    } else {
36        ok(
37            it,
38            Command::new("sh").arg("-c").arg(command),
39            success,
40            failure,
41        )
42    }
43}

45///
46/// Generate man pages
47///
48/// # Errors
49/// On bad user inputs
50///
51#[must_use]
52pub fn generate_man() -> ExitCode {
53    let config = load_config();
54    let len = config.documentation.man.len();
55    for (i, command) in config.documentation.man.iter().enumerate() {
56        if describe(
57            format!("Building Man pages {}/{len}", i + 1).as_str(),
58            command.as_str(),
59            format!("Man step {}/{len} has been completed successfully", i + 1).as_str(),
60            format!("Man step {}/{len} has failed", i + 1).as_str(),
61        )
62        .is_err()
63        {
64            return ExitCode::FAILURE;
65        }
66    }
67    ExitCode::SUCCESS
68}
